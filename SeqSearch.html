Sequential Search: Vector.

#include<iostream>
#include<vector>
#include<string>
using namespace std;

template <typename T>
int seqSearch(const vector<T>& v, int first, int last, const T& target)
{
    int i = first;
    while(i != last)   
    {
        if(v[i] != target)
        {
            i++;
        }
        else if (v[i] == target)
        {
            return i;
        }
    }
return -1;
}
