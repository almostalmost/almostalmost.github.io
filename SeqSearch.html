//template for locating an element in a sequence by it's index.

template 

int seqSearch(const vector& v, int first, int last, const T& target)

{

    int i = first;//initial indexing variable 

    while(i != last)//continue loop as long as i is within desired range "last"     

    {

        if(v[i] != target)//condition checks to see if taget is not found

        {

            i++;//if its not, it checks the next index 

        }

        else if (v[i] == target)//condition checks if target is found

        {

            return i;//if it is, it returns target index

        }

    }

return -1;//if target is not found or is out of range, returns -1

}
