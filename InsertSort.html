Insertion Sort

#include<iostream>
#include<vector>
#include<random>
#include<stdlib.h>
using namespace std;
int main()
{
    int n;
    cout << "insert size of vector: \n";
    cin >> n;
    vector<int>v(n);
    int k;
    for(k=0; k < n; ++k){
    v[k] = rand()%1000000+1;
    }

    //insertion sort
    int i;
    int j;
    int elem;
    for (i = 1; i < n; i++)
    {
        for (elem = v[i], j = i-1; j >=0 and elem < v[j]; j--)
            v[j+1] = v[j];
        v[j+1] = elem;
    }
}
